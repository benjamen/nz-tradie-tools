import os, json, datetime
top10_dir = '/home/ben/.openclaw/workspace/site/data/top10/'
trades_data = json.load(open('/home/ben/.openclaw/workspace/site/data/trades.json'))
cities_data = json.load(open('/home/ben/.openclaw/workspace/site/data/cities.json'))
trades = [t['slug'] for t in trades_data[:12]]
cities = [c['slug'] for c in cities_data]  # All 20 NZ cities
today = datetime.date.today()
missing = []
stale = []  # verified=false or no source_url
for trade in trades:
    for city in cities:
        fname = f'{trade}-{city}.json'
        fpath = os.path.join(top10_dir, fname)
        if not os.path.exists(fpath):
            missing.append((trade, city, fname, 'missing'))
        else:
            data = json.load(open(fpath))
            unverified = [b for b in data.get('businesses', []) if not b.get('verified') or not b.get('source_url')]
            if unverified:
                stale.append((trade, city, fname, f'{len(unverified)} unverified'))
print(f'Missing: {len(missing)} combos')
print(f'Stale (unverified businesses): {len(stale)} combos')
for t,c,f,reason in (missing + stale)[:6]:
    print(f'  {f} ({reason})')
"
