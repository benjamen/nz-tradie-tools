import sys, os
sys.path.insert(0, '/home/ben/polymarket-scanner')
from telegram_bot import send_message

today = __import__('datetime').date.today().isoformat()
filepath = f'/tmp/newsletter-{today}.txt'
content = open(filepath).read()

# Header message
send_message(f'''📬 <b>NZ Tradie Tools — Weekly Newsletter Draft</b>
Date: {today}
Word count: {len(content.split())} words

Copy the content below into Beehiiv: https://app.beehiiv.com
Create new post → paste → review → send.
――――――――――――――――――――''')

# Split into chunks under 3800 chars
chunk_size = 3800
paragraphs = content.split('\n\n')
current_chunk = ''
part = 1
for para in paragraphs:
    if len(current_chunk) + len(para) + 2 > chunk_size:
        if current_chunk:
            send_message(f'📄 <b>Part {part}:</b>\n\n{current_chunk.strip()}')
            part += 1
            current_chunk = para + '\n\n'
    else:
        current_chunk += para + '\n\n'
if current_chunk.strip():
    send_message(f'📄 <b>Part {part}:</b>\n\n{current_chunk.strip()}')
