# Count unique trades and cities correctly (split on FIRST hyphen only,
# so 'builders-lower-hutt' → trade='builders', city='lower-hutt')
trades_with_data = len(set(f.stem.split('-', 1)[0] for f in top10_files))

# Most recent article — read title and date from frontmatter
art_dates = []
for a in articles:
    text = a.read_text()
    title = None
    date = None
    in_front = False
    for line in text.splitlines():
        if line.strip() == '---':
            in_front = not in_front
            continue
        if in_front:
            if line.startswith('title:'):
                title = line.replace('title:', '').strip().strip('\"')
            elif line.startswith('date:'):
                date = line.replace('date:', '').strip()
    if date:
        art_dates.append((date, title or a.stem.replace('-', ' ').title()))
art_dates.sort(reverse=True)
latest_title = art_dates[0][1] if art_dates else 'unknown'
latest_date = art_dates[0][0] if art_dates else 'unknown'

now = datetime.datetime.now()

def days_ago(ts):
    if not ts:
        return 'never'
    d = datetime.datetime.fromtimestamp(ts).date()
    diff = (today - d).days
    if diff == 0: return 'today'
    if diff == 1: return 'yesterday'
